CREATE PROCEDURE [dbo].[sp_Up_UserProfile]
	  @UserId int,
      @Name varchar(50),
      @Email varchar(50),
      @ContactNumber varchar(15),
      @BloodGroup varchar(3),
      @MaritialStatus varchar(15),
      @Designation varchar(50),
      @Department varchar(50),
      @PresentAddress varchar(max),
      @PermanentAddress varchar(max),
      @EmergencyContact varchar(15),
      @JoiningDate varchar(20),
      @Nid varchar(15)
AS
BEGIN


	UPDATE [dbo].[UserLogin]
	   SET 
		  [LoginName] = @Name
	 WHERE  UserId = @UserId



	UPDATE [dbo].[UsersProfile]
	   SET [Name] = @Name
		  ,[Email] = @Email
		  ,[Phone] = @ContactNumber
		  ,[BloodGroup] = @BloodGroup
		  ,[MaritialStatus] = @MaritialStatus
		  ,[Designation] = @Designation
		  ,[Department] = @Department
		  ,[PresentAddress] = @PresentAddress
		  ,[PermanentAddress] = @PermanentAddress
		  ,[EmergencyContact] = @EmergencyContact
		  ,[JoiningDate] = CONVERT(DATE, @JoiningDate,103)
		  ,[Nid] = @Nid
		  ,[Status] = 1
	 WHERE UserId = @UserId


END
